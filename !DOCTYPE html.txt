<body>
    <div class="chat-container">
        <div class="header">Chat with Friends</div>
        <div id="chatbox" class="chatbox">
            <!-- Messages will appear here -->
        </div>
        <div class="input-container">
            <label id="fileLabel" for="fileInput">Upload File</label>
            <input type="file" id="fileInput" />
            <input type="text" id="messageInput" placeholder="Type a message..." />
            <button id="sendButton">Send</button>
        </div>
    </div>
    <script>
        const messageInput = document.getElementById('messageInput');
        const chatbox = document.getElementById('chatbox');
        const sendButton = document.getElementById('sendButton');
        const fileInput = document.getElementById('fileInput');
        const fileLabel = document.getElementById('fileLabel');

        function addMessage(text, sender) {
            const newMessage = document.createElement('div');
            newMessage.classList.add('message', sender);
            newMessage.textContent = text;
            chatbox.appendChild(newMessage);
        }

        sendButton.addEventListener('click', function() {
            if (messageInput.value.trim() !== '') {
                addMessage(messageInput.value, 'you');
                messageInput.value = '';
                chatbox.scrollTop = chatbox.scrollHeight;
            }
        });

        messageInput.addEventListener('keypress', function(e) {
            if (e.key === 'Enter' && messageInput.value.trim() !== '') {
                addMessage(messageInput.value, 'you');
                messageInput.value = '';
                chatbox.scrollTop = chatbox.scrollHeight;
            }
        });

        fileInput.addEventListener('change', function() {
            if (fileInput.files.length > 0) {
                const file = fileInput.files[0];
                const fileMessage = document.createElement('div');
                fileMessage.classList.add('message', 'you');
                fileMessage.textContent = `File: ${file.name}`;
                chatbox.appendChild(fileMessage);
                fileInput.value = '';
                chatbox.scrollTop = chatbox.scrollHeight;
            }
        });

        // Example messages
        addMessage('Hello!', 'them');
        addMessage('Hi there!', 'you');
    </script>
