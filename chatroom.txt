    const herName = "Manpreet Kaur";

    // hearts animation
    const heartsContainer = document.getElementById("hearts");
    function createHeart(){
      const h = document.createElement("div");
      h.className = "heart";
      h.textContent = "‚ù§";
      const size = Math.random()*18+10;
      h.style.fontSize = size + "px";
      h.style.left = Math.random()*100 + "vw";

      h.style.animationDuration = (Math.random()*3+4) + "s";
      heartsContainer.appendChild(h);
      setTimeout(()=>h.remove(),7000);
    }
    let heartInterval = setInterval(createHeart,800);

    // page navigation
    const pages = document.querySelectorAll("section.page");
    function showPage(n){
      pages.forEach(p=>{
        p.classList.remove("active");
      });
      const target = document.getElementById("page"+n);
      if(target) target.classList.add("active");
      window.scrollTo({top:0,behavior:"smooth"});
    }
    document.addEventListener("click",e=>{
      const btn = e.target.closest("[data-go]");
      if(btn){
        const n = btn.getAttribute("data-go");
        showPage(n);
      }
    });

    // runaway "No" button on page 1
    const noBtnMain = document.getElementById("noBtnMain");
    const yesBtnMain = document.getElementById("yesBtnMain");
    const answerMain = document.getElementById("answerMain");

    function moveNoButton(){
      const parent = noBtnMain.parentElement;
      const parentRect = parent.getBoundingClientRect();
      const btnRect = noBtnMain.getBoundingClientRect();

      const maxX = parentRect.width - btnRect.width - 10;
      const maxY = parentRect.height - btnRect.height - 10;

      const newX = Math.random()*maxX;
      const newY = Math.random()*maxY;

      noBtnMain.style.position = "relative";
      noBtnMain.style.left = newX + "px";
      noBtnMain.style.top  = newY + "px";
    }
    noBtnMain.addEventListener("mouseenter",moveNoButton);
    noBtnMain.addEventListener("click",moveNoButton);

    yesBtnMain.addEventListener("click",()=>{
      answerMain.style.opacity = 0;
      setTimeout(()=>{
        answerMain.textContent =
          "If you said yes, then thank you, " +
          herName +
          ". I promise to always respect you and keep this moment safe in my heart. üíó";
        answerMain.style.opacity = 1;

        answerFinal.textContent =
